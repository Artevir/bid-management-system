# 为招投标软件设计完整公司信息管理模块

我将为你的招投标软件设计一个**完整的公司信息管理模块**，涵盖基础信息管理、文件存储、多维度关联等核心能力，完全适配Next\.js 15 \+ TypeScript \+ Drizzle ORM技术栈，同时结合招投标业务特性补充实用功能。

---

## 一、核心需求拆解

### 1\. 核心功能定位

- **公司基础信息管理**：存储企业核心资质信息，支持增删改查

- **文件资产管理**：统一存储商务资质、人员资质、业绩、合同等文件

- **多维度关联**：公司信息可关联到生成的文档、提示词模板，作为归属标识

- **权限控制**：不同用户可管理不同公司的信息和文件

### 2\. 业务关联设计

- 生成文档时可选择**归属公司**，文档自动关联该公司的基础信息

- 提示词模板可绑定**适用公司**，生成时自动填充对应公司参数

- 文件资产可按公司\+文件类型分类管理，方便招投标文件快速调取

import { db } from '@/lib/db';

// 公司信息管理服务
export class CompanyService {
  // 创建/更新公司信息
  async saveCompany(data: NewCompany & { id?: string }): Promise<Company> {
    const companyId = data.id || uuidv4();
    const now = new Date();
    
    // 处理默认公司逻辑（只能有一个默认公司）
    if (data.isDefault) {
      await db
        .update(companies)
        .set({ isDefault: false, updatedAt: now })
        .where(eq(companies.userId, data.userId));
    }
    
    // 检查是否存在
    const existing = await db
      .select()
      .from(companies)
      .where(eq(companies.id, companyId))
      .limit(1);
    
    const companyData = {
      ...data,
      id: companyId,
      updatedAt: now,
    };
    
    if (existing.length > 0) {
      // 更新
      await db.update(companies).set(companyData).where(eq(companies.id, companyId));
    } else {
      // 创建
      await db.insert(companies).values({
        ...companyData,
        createdAt: now,
        isActive: true,
      });
    }
    
    // 返回最新数据
    const [result] = await db
      .select()
      .from(companies)
      .where(eq(companies.id, companyId))
      .limit(1);
    
    return result;
  }

  // 获取公司详情
  async getCompany(id: string): Promise<Company | null> {
    const [company] = await db
      .select()
      .from(companies)
      .where(and(eq(companies.id, id), eq(companies.isActive, true)))
      .limit(1);
    
    return company || null;
  }

  // 获取公司列表
  async listCompanies(
    userId: string,
    keyword?: string,
    page = 1,
    pageSize = 10
  ): Promise<{ list: Company[]; total: number }> {
    // 构建查询条件
    let query = db
      .select()
      .from(companies)
      .where(and(eq(companies.userId, userId), eq(companies.isActive, true)));
    
    // 关键词搜索
    if (keyword) {
      query = query.where(
        or(
          like(companies.name, `%${keyword}%`),
          like(companies.taxId, `%${keyword}%`),
          like(companies.contactPersonName, `%${keyword}%`)
        )
      );
    }
    
    // 总数
    const totalQuery = db
      .select({ count: sql<number>`count(*)` })
      .from(companies)
      .where(query.where?.conditions || eq(companies.id, sql`*`));
    
    const [totalResult] = await totalQuery;
    const total = totalResult.count;
    
    // 分页查询
    const list = await query
      .orderBy(companies.isDefault, 'desc')
      .orderBy(companies.updatedAt, 'desc')
      .limit(pageSize)
      .offset((page - 1) * pageSize);
    
    return { list, total };
  }

  // 删除公司（软删除）
  async deleteCompany(id: string): Promise<boolean> {
    const result = await db
      .update(companies)
      .set({ isActive: false, updatedAt: new Date() })
      .where(eq(companies.id, id));
    
    return result.rowCount > 0;
  }

  // 获取默认公司
  async getDefaultCompany(userId: string): Promise<Company | null> {
    const [company] = await db
      .select()
      .from(companies)
      .where(and(eq(companies.userId, userId), eq(companies.isDefault, true), eq(companies.isActive, true)))
      .limit(1);
    
    return company || null;
  }
}

export class CompanyFileService {
  private storageService: StorageService;
  
  constructor() {
    this.storageService = new StorageService(); // 初始化存储服务
  }

  // 上传公司文件
  async uploadCompanyFile(
    companyId: string,
    file: File,
    fileType: FileType,
    fileName: string,
    description: string | null,
    uploaderId: string
  ): Promise<CompanyFile> {
    // 1. 上传文件到存储
    const fileKey = `company/${companyId}/${fileType}/${uuidv4()}-${file.name}`;
    const fileUrl = await this.storageService.uploadFile(file, fileKey);
    
    // 2. 计算文件MD5（可选）
    const fileMd5 = await this.calculateFileMd5(file);
    
    // 3. 保存文件信息到数据库
    const fileId = uuidv4();
    const fileData: NewCompanyFile = {
      id: fileId,
      companyId,
      fileName: fileName || file.name,
      fileType,
      fileUrl,
      fileSize: this.formatFileSize(file.size),
      fileExt: file.name.split('.').pop() || '',
      fileMd5,
      description: description || null,
      uploaderId,
      isActive: true,
      createdAt: new Date(),
      updatedAt: new Date(),
    };
    
    await db.insert(companyFiles).values(fileData);
    
    // 4. 返回文件信息
    const [result] = await db
      .select()
      .from(companyFiles)
      .where(eq(companyFiles.id, fileId))
      .limit(1);
    
    return result;
  }

  // 获取公司文件列表
  async listCompanyFiles(
    companyId: string,
    fileType?: FileType,
    keyword?: string
  ): Promise<CompanyFile[]> {
    let query = db
      .select()
      .from(companyFiles)
      .where(and(eq(companyFiles.companyId, companyId), eq(companyFiles.isActive, true)));
    
    // 文件类型筛选
    if (fileType) {
      query = query.where(eq(companyFiles.fileType, fileType));
    }
    
    // 关键词搜索
    if (keyword) {
      query = query.where(like(companyFiles.fileName, `%${keyword}%`));
    }
    
    return query.orderBy(companyFiles.createdAt, 'desc');
  }

  // 删除公司文件
  async deleteCompanyFile(id: string): Promise<boolean> {
    // 1. 获取文件信息
    const [file] = await db
      .select()
      .from(companyFiles)
      .where(eq(companyFiles.id, id))
      .limit(1);
    
    if (!file) return false;
    
    // 2. 软删除数据库记录
    await db
      .update(companyFiles)
      .set({ isActive: false, updatedAt: new Date() })
      .where(eq(companyFiles.id, id));
    
    // 3. 删除存储文件（可选，可保留备份）
    // await this.storageService.deleteFile(file.fileUrl);
    
    return true;
  }

  // 格式化文件大小

    if (bytes < 1024) return `${bytes} B`;
    if (bytes < 1024 * 1024) return `${(bytes / 1024).toFixed(2)} KB`;
    return `${(bytes / (1024 * 1024)).toFixed(2)} MB`;

// 文件类型映射
const fileTypeMap: Record<FileType, { label: string; icon: React.ReactNode }> = {

  other: { label: '其他文件', icon: <FileText className="h-4 w-4" /> },
};

export function CompanyFileUploader({
  companyId,
  onUploadComplete,
  className,
}: CompanyFileUploaderProps) {
  const [fileType, setFileType] = useState<FileType>('other');
  const [fileName, setFileName] = useState('');
  const [description, setDescription] = useState('');
  const [selectedFile, setSelectedFile] = useState<File | null>(null);
  const [isUploading, setIsUploading] = useState(false);
  const fileInputRef = useRef<HTMLInputElement>(null);

  // 处理文件选择
  const handleFileChange = (e: React.ChangeEvent<HTMLInputElement>) => {
    const files = e.target.files;
    if (files && files.length > 0) {
      const file = files[0];
      setSelectedFile(file);
      if (!fileName) {
        setFileName(file.name.replace(/\.[^/.]+$/, '')); // 去除扩展名
      }
      
      // 根据文件类型自动匹配
      const ext = file.name.split('.').pop()?.toLowerCase();
      if (['pdf', 'jpg', 'png'].includes(ext || '')) {
        if (file.name.includes('营业执照')) setFileType('business_license');
        else if (file.name.includes('资质')) setFileType('business_certificate');
        else if (file.name.includes('业绩')) setFileType('performance_scan');
        else if (file.name.includes('合同')) setFileType('contract');
      }
    }
  };

  // 触发文件选择
  const triggerFileSelect = () => {
    fileInputRef.current?.click();
  };

  // 处理文件上传
  const handleUpload = async () => {
    if (!companyId || !selectedFile) {
      toast.error('请选择要上传的文件');
      return;
    }

    setIsUploading(true);
    
    try {
      // 创建FormData
      const formData = new FormData();
      formData.append('companyId', companyId);
      formData.append('fileType', fileType);
      formData.append('fileName', fileName || selectedFile.name);
      formData.append('description', description);
      formData.append('file', selectedFile);

      // 调用上传API
      const response = await fetch(`/api/companies/${companyId}/files`, {
        method: 'POST',
        body: formData,
      });

      if (response.ok) {
        const data = await response.json();
        toast.success('文件上传成功');
        
        // 重置状态
        setSelectedFile(null);
        setFileName('');
        setDescription('');
        setFileType('other');
        if (fileInputRef.current) fileInputRef.current.value = '';
        
        // 回调
        if (onUploadComplete) {
          onUploadComplete(data.data);
        }
      } else {
        throw new Error('文件上传失败');
      }
    } catch (error) {
      toast.error((error as Error).message || '文件上传失败');
    } finally {
      setIsUploading(false);
    }
  };

  // 文件大小校验
  const isFileTooLarge = selectedFile && selectedFile.size > 10 * 1024 * 1024; // 10MB

  return (
    <Card className={cn(className)}>
      <CardHeader>
        <CardTitle>文件上传</CardTitle>
        <CardDescription>上传公司相关资质文件，单文件最大10MB</CardDescription>
      </CardHeader>
      <CardContent className="space-y-4">
        <input
          type="file"
          ref={fileInputRef}
          onChange={handleFileChange}
          className="hidden"
          accept=".pdf,.jpg,.jpeg,.png,.doc,.docx,.xls,.xlsx"
        />

        {/* 文件选择 */}
        <div 
          className="border-2 border-dashed rounded-md p-8 text-center hover:border-primary transition-colors cursor-pointer"
          onClick={triggerFileSelect}
        >
          <FileUpload className="mx-auto h-12 w-12 text-muted-foreground mb-4" />
          <p className="text-sm text-muted-foreground mb-2">
            点击或拖拽文件到此处上传
          </p>
          <p className="text-xs text-muted-foreground">
            支持 PDF, JPG, PNG, DOC, DOCX, XLS, XLSX 格式
          </p>
          
          {selectedFile && (
            <div className="mt-4 p-3 bg-muted rounded-md text-left">
              <div className="flex items-center justify-between mb-2">
                <span className="font-medium text-sm">{selectedFile.name}</span>
                <span className="text-xs text-muted-foreground">
                  {(selectedFile.size / 1024 / 1024).toFixed(2)} MB

              </div>
              
              {isFileTooLarge && (
                <div className="flex items-center text-destructive text-xs mt-2">
                  <AlertCircle className="h-3 w-3 mr-1" />
                  文件大小超过10MB限制，请压缩后上传
                </div>
              )}
            </div>
          )}
        </div>

        {/* 文件类型 */}
        <Select value={fileType} onValueChange={(value) => setFileType(value as FileType)}>
