1. Next.js 15 动态路由参数
错误模式
export async function GET(  request: NextRequest,  { params }: { params: { id: string } }) {  const { id } = params;}
正确模式
export async function GET(  request: NextRequest,  { params }: { params: Promise<{ id: string }> }) {  const { id } = await params;}
适用对象
[id]/route.ts 
[roleId]/route.ts 
[tenantId]/route.ts 
[sessionId]/route.ts 
2. 中间件 async 返回函数调用
错误模式
return withDocumentPermission('read', () => documentId)(request, handler);
正确模式
const middleware = await withDocumentPermission('read', () => documentId);return middleware(request, handler);
同类适用
withDocumentPermission(...) 
withChapterPermission(...) 
withResourcePermission(...) 
你已经在 src/app/api/bid/chapters/route.ts 上遇到过这一类。

4. schema 导入路径漂移
错误模式
import { projects } from '@/db/schema/projects';import { documents } from '@/db/schema/documents';import { companies } from '@/db/schema/companies';import { auditLogs } from '@/db/schema/audit-logs';
正确模式
import { projects, bidDocuments, companies } from '@/db/schema';import { auditLogs } from '@/lib/db/schema/audit-logs';

错误模式
documents
正确模式

5. auth 模块调用口径
错误模式
import { requireAuth } from '@/lib/auth/middleware';const authResult = await requireAuth(request);
正确模式 A：如果走 session 口径
import { requireAuth } from '@/lib/auth/session';const user = await requireAuth();
正确模式 B：如果走 middleware 口径
import { requireAuth } from '@/lib/auth/middleware';const authResult = await requireAuth(request);
规则
不要混用。要么这类路由统一走 session.requireAuth()，要么统一走 middleware.requireAuth(request)。
你现在仓库里这两个版本都存在，所以后续最好选一套。
6. withAdmin 旧封装
错误模式
return withAdmin(request, createUser);
正确模式
return withAuth(request, async (req, userId, params) => {  const ok = await checkPermission(userId, '真实管理员权限码');  if (!ok) {    return NextResponse.json({ error: '需要管理员权限' }, { status: 403 });  }  return createUser(req, userId, params);});
说明
这是长期最干净的方式。不过如果你当前 middleware.ts 里已经补回了 withAdmin，那这条可以作为后续重构目标，不一定非要立刻改完。
7. cache 导入口径
错误模式
import { cache } from '@/lib/cache';import redis, { RedisKeys } from '@/lib/cache';import { cacheGet, cacheSet } from '@/lib/cache';
正确模式 A：底层 Redis 能力
import redis, {  RedisKeys,  getRedisClient,  isUsingMemoryFallback,  cacheGet,  cacheSet,  cacheDelete,  cacheDeletePattern,  cacheExists,  cacheTTL,  invalidateCompanyCache,} from '@/lib/cache/redis-client';
正确模式 B：旧 cache 变量写法
import { redis as cache } from '@/lib/cache/redis-client';
规则
业务层缓存聚合逻辑，才从 @/lib/cache 
直接用 Redis 客户端能力，就从 @/lib/cache/redis-client 
你前面的 build 和 grep 已经证明这块漂移很多。
8. LLM 调用方式
错误模式
const llm = getLLM();const response = await llm.chat({ messages });const stream = await llm.chatStream({ messages });
正确模式
const content = await invokeChat({ messages }, forwardHeaders);const stream = streamChat({ messages }, forwardHeaders);
规则
流式统一走 streamChat(...) 
非流式统一走 invokeChat(...) 
不直接依赖 adapter 实例上的 chat / chatStream 
因为你已经确认 LLMAdapter 和调用方不一致，而 service 层函数是稳定存在的。

const ipAddress =  request.headers.get('x-forwarded-for')?.split(',')[0]?.trim() ||  request.headers.get('x-real-ip') ||  request.headers.get('cf-connecting-ip') ||  'unknown';

11. 空数组被推断成 never[]
错误模式
const [options, setOptions] = useState({  templates: [],  companies: [],  availableTags: [],});
正确模式 A：显式 state 类型
type OptionsState = {  templates: { id: number; name: string }[];  companies: { id: number; name: string }[];  availableTags: { id: number; name: string }[];};const [options, setOptions] = useState<OptionsState>({  templates: [],  companies: [],  availableTags: [],});
正确模式 B：最小断言
const [options, setOptions] = useState({  templates: [] as { id: number; name: string }[],  companies: [] as { id: number; name: string }[],  availableTags: [] as { id: number; name: string }[],});
你在编辑页已经碰到过这一类。

13. const 被重新赋值
错误模式
const startDate = parsedStartDate;const endDate = parsedEndDate;if (!startDate && !endDate) {  startDate = thirtyDaysAgo;  endDate = now;}
正确模式 A
let startDate = parsedStartDate;let endDate = parsedEndDate;if (!startDate && !endDate) {  startDate = thirtyDaysAgo;  endDate = now;}
正确模式 B：更推荐
const now = new Date();const thirtyDaysAgo = new Date(now.getTime() - 30 * 24 * 60 * 60 * 1000);const startDate = parsedStartDate || thirtyDaysAgo;const endDate = parsedEndDate || now;
